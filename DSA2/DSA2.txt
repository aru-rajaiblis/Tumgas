PROGRAM Sistem_Seleksi_Magang

DEKLARASI
    N, i : INTEGER
    jml_valid, jml_tidak_valid, jml_lulus : INTEGER
    total_nilai, rata_rata : FLOAT
    max1, max2 : FLOAT // BARU: Deklarasi variabel untuk dua nilai tertinggi
    
    // Asumsi array nama, tes_tulis, dan wawancara sudah berisi data pelamar
    nama : ARRAY OF STRING
    tes_tulis : ARRAY OF FLOAT
    wawancara : ARRAY OF FLOAT
    
    // Array untuk menyimpan hasil proses
    nilai_akhir : ARRAY OF FLOAT
    status : ARRAY OF STRING

ALGORITMA
    // Inisialisasi variabel
    jml_valid <- 0
    jml_tidak_valid <- 0
    jml_lulus <- 0
    total_nilai <- 0
    rata_rata <- 0
    max1 <- 0
    max2 <- 0 

    // MENGHITUNG ISI ARRAY: Menentukan N berdasarkan panjang/banyaknya data di array 'nama'
    N <- LENGTH(nama) 

    // Proses Data
    FOR i <- 0 TO N - 1 DO 
        
        // Validasi data: Rentang 0 - 100
        IF (tes_tulis[i] < 0 OR tes_tulis[i] > 100) OR (wawancara[i] < 0 OR wawancara[i] > 100) THEN
            status[i] <- "DATA TIDAK VALID"
            nilai_akhir[i] <- 0
            jml_tidak_valid <- jml_tidak_valid + 1
        ELSE
            // Kalau Data Valid
            jml_valid <- jml_valid + 1
            nilai_akhir[i] <- (0.6 * tes_tulis[i]) + (0.4 * wawancara[i])
            total_nilai <- total_nilai + nilai_akhir[i]
            
            // Penentuan Status Kelulusan
            IF nilai_akhir[i] >= 75 THEN
                status[i] <- "LULUS"
                jml_lulus <- jml_lulus + 1
                
                // BARU: Logika mencari dua nilai tertinggi yang LULUS (tanpa sorting)
                IF nilai_akhir[i] > max1 THEN
                    max2 <- max1
                    max1 <- nilai_akhir[i]
                ELSE IF (nilai_akhir[i] < max1) AND (nilai_akhir[i] > max2) THEN
                    max2 <- nilai_akhir[i]
                END IF
            ELSE
                status[i] <- "TIDAK LULUS"
            END IF
        END IF
    END FOR

    // Menghitung Rata-rata Nilai Akhir dengan data yang valid
    IF jml_valid > 0 THEN
        rata_rata <- total_nilai / jml_valid
    ELSE
        rata_rata <- 0
    END IF

    // Output
    PRINT "=== DAFTAR HASIL PELAMAR ==="
    FOR i <- 0 TO N - 1 DO // REVISI KECIL: Diubah jadi N - 1
        IF status[i] = "DATA TIDAK VALID" THEN
            PRINT nama[i], " | Nilai Akhir: - | Status: ", status[i]
        ELSE
            PRINT nama[i], " | Nilai Akhir: ", nilai_akhir[i], " | Status: ", status[i]
        END IF
    END FOR

    PRINT "=== STATISTIK ==="
    PRINT "Jumlah Data Valid:", jml_valid
    PRINT "Jumlah Data Tidak Valid:", jml_tidak_valid
    PRINT "Jumlah Pelamar Lulus:", jml_lulus
    PRINT "Rata-rata Nilai Akhir (Data Valid):", rata_rata
    
    // BARU: Menampilkan output dua nilai tertinggi
    IF jml_lulus >= 2 THEN
        PRINT "Dua Nilai Tertinggi (LULUS): ", max1, " dan ", max2
    ELSE
        PRINT "Dua Nilai Tertinggi (LULUS): Tidak ada pelamar yang lulus."
    END IF
    
END PROGRAM
